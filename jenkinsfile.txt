pipeline {
    agent any

    stages {
        stage('Git pull') {
            steps {
                git branch: 'main', url: 'https://github.com/anuragsroy/system-data-python.git'
            }
        }
        stage('Docker image backend') {
            steps {
                bat 'docker build -t python-backend-image .'
            }
        }
         stage('Docker container backend') {
            steps {
                bat 'docker run -d -p 5000:5000 --name python-backend-container python-backend-image'
            }
        }
    
        stage('Docker image frontend') {
            steps {
                dir('frontend'){
                bat 'docker build -t js-frontend-image .'
                }
            }
        }
         stage('Docker container frontend') {
            steps {
                bat 'docker run -d -p 5050:5050 --name js-frontend-container js-frontend-image'
            }
        }
    }
}
